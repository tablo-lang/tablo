        set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} -fsanitize=address,undefined")
    endif()
endif()

if(TABLO_ENABLE_GLFW_EXTENSION)
    find_package(glfw3 QUIET CONFIG)
    if(TARGET glfw)
        set(TABLO_GLFW_TARGET glfw)
    elseif(TARGET glfw3::glfw)
        set(TABLO_GLFW_TARGET glfw3::glfw)
    elseif(TARGET glfw3)
        set(TABLO_GLFW_TARGET glfw3)
    elseif(TABLO_FETCH_GLFW)
        include(FetchContent)
        set(GLFW_BUILD_DOCS OFF CACHE BOOL "" FORCE)
        set(GLFW_BUILD_TESTS OFF CACHE BOOL "" FORCE)
        set(GLFW_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)
        set(GLFW_INSTALL OFF CACHE BOOL "" FORCE)
        FetchContent_Declare(
            glfw
            GIT_REPOSITORY https://github.com/glfw/glfw.git
            GIT_TAG 3.4
        )
        FetchContent_MakeAvailable(glfw)
        set(TABLO_GLFW_TARGET glfw)
    endif()

    if(NOT TABLO_GLFW_TARGET)
        message(FATAL_ERROR "TABLO_ENABLE_GLFW_EXTENSION=ON requires glfw3. Set CMAKE_PREFIX_PATH/GLFW3_DIR or enable TABLO_FETCH_GLFW.")
    endif()

    add_library(tablo_glfw_extension SHARED tests/tablo_glfw_extension.c)
    target_include_directories(tablo_glfw_extension PRIVATE ${PROJECT_SOURCE_DIR}/src)
    target_link_libraries(tablo_glfw_extension PRIVATE ${TABLO_GLFW_TARGET} Threads::Threads)
    set_target_properties(tablo_glfw_extension PROPERTIES PREFIX "")

    add_executable(glfw_extension_host_tests tests/glfw_extension_host_tests.c)
    target_link_libraries(glfw_extension_host_tests tablo_lib)
    if(UNIX AND NOT APPLE)
        target_link_libraries(glfw_extension_host_tests ${CMAKE_DL_LIBS})
    endif()

if(ENABLE_PERF_GATES OR ENABLE_FAST_STRING_BASELINE_GATE)
    find_package(Python3 COMPONENTS Interpreter QUIET)
    if(Python3_Interpreter_FOUND)
        set(SHOULD_ADD_RELEASE_PERF_TESTS TRUE)
        if(CMAKE_CONFIGURATION_TYPES)
            # Perf gates are calibrated for optimized builds only.
            set(PERF_GATE_CONFIG "Release")
        else()
            set(PERF_GATE_CONFIG "Release")
            if(CMAKE_BUILD_TYPE)
                set(PERF_GATE_CONFIG "${CMAKE_BUILD_TYPE}")
            endif()
            string(TOLOWER "${PERF_GATE_CONFIG}" PERF_GATE_CONFIG_LOWER)
            if(NOT PERF_GATE_CONFIG_LOWER STREQUAL "release")
                set(SHOULD_ADD_RELEASE_PERF_TESTS FALSE)
                if(ENABLE_PERF_GATES)
                    message(STATUS "ENABLE_PERF_GATES=ON but non-Release single-config build (${PERF_GATE_CONFIG}); skipping PerfGates test.")
                endif()
                if(ENABLE_FAST_STRING_BASELINE_GATE)
                    message(STATUS "ENABLE_FAST_STRING_BASELINE_GATE=ON but non-Release single-config build (${PERF_GATE_CONFIG}); skipping FastStringBaselineGate test.")
                endif()
            endif()
        endif()

        if(SHOULD_ADD_RELEASE_PERF_TESTS)
            if(ENABLE_PERF_GATES)
                add_test(
                    NAME PerfGates
                    CONFIGURATIONS Release
                    COMMAND ${Python3_EXECUTABLE} ${CMAKE_CURRENT_SOURCE_DIR}/tools/check_perf_gates.py --config ${PERF_GATE_CONFIG} --samples 5 --warmup-runs 1
                    WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
                )
                set_tests_properties(PerfGates PROPERTIES LABELS "perf" RUN_SERIAL TRUE)
            endif()

            if(ENABLE_FAST_STRING_BASELINE_GATE)
                if(NOT FAST_STRING_BASELINE_FILE)
                    message(FATAL_ERROR "ENABLE_FAST_STRING_BASELINE_GATE=ON requires FAST_STRING_BASELINE_FILE to be set.")
                endif()

                set(FAST_STRING_BASELINE_PATH "${FAST_STRING_BASELINE_FILE}")
                if(IS_ABSOLUTE "${FAST_STRING_BASELINE_PATH}" AND NOT EXISTS "${FAST_STRING_BASELINE_PATH}")
                    file(RELATIVE_PATH FAST_STRING_BASELINE_RELATIVE_FROM_BINARY
                         "${CMAKE_CURRENT_BINARY_DIR}"
                         "${FAST_STRING_BASELINE_PATH}")
                    if(NOT FAST_STRING_BASELINE_RELATIVE_FROM_BINARY MATCHES "^\\.\\.")
                        set(FAST_STRING_BASELINE_SOURCE_CANDIDATE
                            "${CMAKE_CURRENT_SOURCE_DIR}/${FAST_STRING_BASELINE_RELATIVE_FROM_BINARY}")
                        if(EXISTS "${FAST_STRING_BASELINE_SOURCE_CANDIDATE}")
                            set(FAST_STRING_BASELINE_PATH "${FAST_STRING_BASELINE_SOURCE_CANDIDATE}")
                            set(FAST_STRING_BASELINE_FILE
                                "${FAST_STRING_BASELINE_RELATIVE_FROM_BINARY}"
                                CACHE STRING
                                "Path to baseline JSON generated by tools/benchmark_fast_string_baseline.py"
                                FORCE)
                        endif()
                    endif()
                endif()
                if(NOT IS_ABSOLUTE "${FAST_STRING_BASELINE_PATH}")
                    set(FAST_STRING_BASELINE_PATH "${CMAKE_CURRENT_SOURCE_DIR}/${FAST_STRING_BASELINE_PATH}")
                endif()
                get_filename_component(FAST_STRING_BASELINE_PATH "${FAST_STRING_BASELINE_PATH}" ABSOLUTE)
                if(NOT EXISTS "${FAST_STRING_BASELINE_PATH}")
                    message(FATAL_ERROR "FAST_STRING_BASELINE_FILE does not exist: ${FAST_STRING_BASELINE_PATH}")
                endif()

                add_test(
                    NAME FastStringBaselineGate
                    CONFIGURATIONS Release
                    COMMAND ${Python3_EXECUTABLE}
                            ${CMAKE_CURRENT_SOURCE_DIR}/tools/check_fast_string_regression.py
                            --config ${PERF_GATE_CONFIG}
                            --baseline ${FAST_STRING_BASELINE_PATH}
                            --scale ${FAST_STRING_BASELINE_SCALE}
                            --samples ${FAST_STRING_BASELINE_SAMPLES}
                            --warmup-runs ${FAST_STRING_BASELINE_WARMUP_RUNS}
                            --transient-retries ${FAST_STRING_BASELINE_TRANSIENT_RETRIES}
                            --max-metric-regression-pct ${FAST_STRING_MAX_METRIC_REGRESSION_PCT}
                            --max-speedup-regression-pct ${FAST_STRING_MAX_SPEEDUP_REGRESSION_PCT}
                            --candidate-out ${CMAKE_CURRENT_BINARY_DIR}/fast_string_candidate_${PERF_GATE_CONFIG}.json
                            --compare-out ${CMAKE_CURRENT_BINARY_DIR}/fast_string_compare_${PERF_GATE_CONFIG}.json
                    WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
                )
                set_tests_properties(FastStringBaselineGate PROPERTIES LABELS "perf" RUN_SERIAL TRUE)
            endif()
        endif()
    else()
        if(ENABLE_PERF_GATES)
            message(WARNING "ENABLE_PERF_GATES=ON but Python3 interpreter was not found; skipping PerfGates CTest.")
        endif()
        if(ENABLE_FAST_STRING_BASELINE_GATE)
            message(WARNING "ENABLE_FAST_STRING_BASELINE_GATE=ON but Python3 interpreter was not found; skipping FastStringBaselineGate CTest.")
        endif()
    endif()
endif()
